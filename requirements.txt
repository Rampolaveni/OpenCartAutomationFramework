# -----------------------------------------------------------------------------
# Parallel execution
# -----------------------------------------------------------------------------

pytest-xdist==3.8.0
# Runs tests across multiple CPU cores simultaneously.
# Usage: pytest -n auto         (auto-detect cores)
#        pytest -n 4            (fixed 4 workers)
# Each worker gets its own Browser + BrowserContext — no shared state.
# Requires tests to be fully independent (no shared DB user, no hardcoded IDs).

allure-pytest==2.16.0
# Rich interactive reports with steps, attachments, and history trends.
#
# Usage:
#   pytest --alluredir=allure-results
#   allure serve allure-results         (opens report in browser)
#   allure generate allure-results -o allure-report  (static export)
#
# Install Allure CLI separately:
#   brew install allure                 (Mac)
#   scoop install allure               (Windows)
#
# Decorators available in tests:
#   @allure.feature("Authentication")
#   @allure.story("Login")
#   @allure.severity(allure.severity_level.CRITICAL)
#   with allure.step("Fill login form"): ...


# -----------------------------------------------------------------------------
# Retry flaky tests
# -----------------------------------------------------------------------------

pytest-rerunfailures==16.3
# Automatically retries failing tests before marking them as failed.
# Usage: pytest --reruns 2 --reruns-delay 1
# Per-test: @pytest.mark.flaky(reruns=3)
# In pytest.ini: addopts = --reruns 1
# Best practice: fix flaky tests rather than masking them — but useful
# for known infrastructure instability (network timeouts, slow CI agents).
